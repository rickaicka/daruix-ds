@use 'sass:map';
@use '../tokens/breakpoints' as *;

@function drx-breakpoint($size) {
  @if map.has-key($drx-breakpoints, $size) {
    @return map.get($drx-breakpoints, $size);
  }

  @error 'Breakpoint "#{$size}" não encontrado.';
}

@mixin media-down($size) {
  @media (max-width: drx-breakpoint($size)) {
    @content;
  }
}

@mixin media-up($size) {
  @media (min-width: drx-breakpoint($size)) {
    @content;
  }
}

@mixin media-between($min, $max) {
  @media (min-width: drx-breakpoint($min)) and (max-width: drx-breakpoint($max)) {
    @content;
  }
}

// Legacy alias kept for existing screens.
@mixin respond($breakpoint) {
  @if $breakpoint == micro {
    @include media-down(micro) {
      @content;
    }
  } @else if $breakpoint == mobile {
    @include media-down(mobile-md) {
      @content;
    }
  } @else if $breakpoint == tablet {
    @include media-down(tablet) {
      @content;
    }
  } @else if $breakpoint == notebook {
    @include media-down(notebook) {
      @content;
    }
  } @else if $breakpoint == desktop {
    @include media-up(desktop) {
      @content;
    }
  } @else {
    @error 'Breakpoint legado "#{$breakpoint}" não encontrado.';
  }
}
